package org.pircbotx;

/**
 * An enum to represent the possible IRC levels a user can have.
 *
 * @author Leon Blakey
 */
public enum UserLevel {
	VOICE("+"),
	HALFOP("%"),
	OP("@"),
	SUPEROP("&"),
	OWNER("~");
	private final String symbol;
	
	/**
	 * Combines all UserLevel's symbols into a String
	 * @return
	 */
	public static String getSymbols() {
		StringBuilder symbols = new StringBuilder();
		for (UserLevel curLevel : values()) symbols.append(curLevel.getSymbol());
		return symbols.toString();
	}
	
	/**
	 * Searches for UserLevel that uses the given symbol
	 * @param symbol
	 * @return Matching UserLevel or null if not found
	 */
	public static UserLevel fromSymbol(char symbol) {
		for (UserLevel curLevel : values()) if (curLevel.getSymbol().contains(String.valueOf(symbol))) return curLevel;
		return null;
	}

		this.symbol = symbol;
	}
