		if (unit == null) {

		}
		while (true) {
			Event curEvent = eventQueue.poll(timeout, unit);
			//When poll times out it returns null. Repeat that behavior here
			if (curEvent == null) return null;
			for (Class<? extends GenericEvent> curEventClass : eventClasses) if (curEventClass.isInstance(curEvent)) return curEvent;
		}
